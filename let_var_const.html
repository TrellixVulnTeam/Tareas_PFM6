Let

Al declarar "let" este funciona solo como una variable local

let valor
let x = 1;
if (x === 1) {
let x = 2;
console.log(x);
// ouput: 2
}
console.log(x);
// ouput: 1
output:

Var

Al declarar "var" este funciona como una variable global

var valor
var x = 1;
if (x === 1) {
var x = 2;
console.log(x);
// ouput: 2
}
console.log(x);
// ouput: 2
output:

Const

Al declarar "const" funciona como una variable que no puede cambiar y no puede ser redeclarado

const valor
const x = 1;
if (x === 1) {
try {
x = 2;
}
catch (error) {
console.log(error);
// ouput: TypeError: invalid assignment to const `X'
}
}
console.log(x);
// ouput: 1
